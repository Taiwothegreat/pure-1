document.addEventListener("DOMContentLoaded", function () { const dropdownTriggers = document.querySelectorAll(".nav > li"); const menuToggle = document.getElementById("menu-toggle"); const nav = document.getElementById("main-nav"); // Hamburger toggle for mobile menuToggle.addEventListener("click", function (e) { e.stopPropagation(); nav.classList.toggle("active"); }); // Dropdown toggle using mouseover instead of click dropdownTriggers.forEach(item => { const hasDropdown = item.querySelector(".dropdown-content"); if (hasDropdown) { item.addEventListener("mouseenter", function () { // Remove active class from all other items dropdownTriggers.forEach(el => { if (el !== item) el.classList.remove("active"); }); // Add active class to current item item.classList.add("active"); }); item.addEventListener("mouseleave", function () { item.classList.remove("active"); }); // Optional: clicking inside dropdown closes it (e.g. on mobile) hasDropdown.addEventListener("click", function () { item.classList.remove("active"); nav.classList.remove("active"); }); } }); // Close dropdowns and menu when clicking outside document.addEventListener("click", function (e) { if (!nav.contains(e.target) && !menuToggle.contains(e.target)) { dropdownTriggers.forEach(item => item.classList.remove("active")); nav.classList.remove("active"); } }); });
